51template <typename Ret, typename... Args>
52vex::thread make_thread(Ret (*fn)(Args...), Args... args) {
53 // Create a boolean value that gets set to true by the thread so that the thread gets initialized before this function exits; vex STL is buggy
54 bool fake_promise = false;
55 // Put the function pointer and all arguments into a tuple
56 auto* func_and_args = new std::tuple<decltype(fn), std::tuple<Args...>, bool*>(fn, std::forward_as_tuple(args...), &fake_promise);
57 // Create a thread that runs the function wrapper
58 vex::thread internal_thread = vex::thread(internal::wrap_func_<decltype(fn), Args...>, (void*)func_and_args);
59 // Wait for thread initialization
60 while (!fake_promise) {
61 vex::task::sleep(1);
62 }
63 return internal_thread;
64}
